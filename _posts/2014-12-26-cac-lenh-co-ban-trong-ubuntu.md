#1. Kiểm tra danh sách các bản cập nhật

{% highlight bash %}
  sudo apt-get update
{% endhighlight %}

#2. Cập nhật, nâng cấp các package

{% highlight bash %}
  sudo apt-get upgrade
{% endhighlight %}

{% highlight bash %}
sudo apt-get dist-upgrade
{% endhighlight %}

> **Tips**: Sử dụng tham số `-y` để tự động confirm.
